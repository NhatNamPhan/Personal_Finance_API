User Endpoints
POST   /users                    # Tạo user mới
GET    /users/{user_id}          # Lấy thông tin user theo ID

Account Endpoints
POST   /accounts                 # Tạo tài khoản mới
GET    /accounts?user_id={user_id}          # Lấy danh sách tài khoản của user

Category Endpoints
POST   /categories               # Tạo danh mục mới
GET    /categories?user_id={user_id}        # Lấy danh sách danh mục của user
GET    /categories?user_id={user_id}&type={type}  # Lọc danh mục theo type
GET    /categories/{category_id} # Lấy chi tiết danh mục
PUT    /categories/{category_id} # Cập nhật danh mục
DELETE /categories/{category_id} # Xóa danh mục

Transaction Endpoints
POST   /transactions             # Thêm giao dịch mới
GET    /transactions?user_id={user_id}      # Lấy lịch sử giao dịch của user
GET    /transactions?user_id={user_id}&start_date=2024-01-01&end_date=2024-01-31  # Lọc theo khoảng thời gian
GET    /transactions/{transaction_id}       # Lấy chi tiết giao dịch

Budget Endpoints
POST   /budgets                  # Tạo ngân sách mới
GET    /budgets?user_id={user_id}           # Lấy danh sách ngân sách của user
GET    /budgets?user_id={user_id}&month=2024-01-01  # Lọc ngân sách theo tháng
GET    /budgets/{budget_id}      # Lấy chi tiết ngân sách
PUT    /budgets/{budget_id}      # Cập nhật ngân sách
DELETE /budgets/{budget_id}      # Xóa ngân sách

Analytics Endpoints
GET    /analytics/users/{user_id}/spending?start_date=2024-01-01&end_date=2024-01-31  # Thống kê chi tiêu
GET    /analytics/users/{user_id}/budgets/progress?month=2024-01-01  # Tiến độ ngân sách
GET    /analytics/users/{user_id}/net-worth           # Tổng tài sản ròng
